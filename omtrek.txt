
function berekenInhoud(lengte, breedte, hoogte) {
    return lengte * breedte * hoogte;
  }
  
  // Voorbeeld aanroep
  let kubusInhoud = berekenInhoud(10, 5, 10);
  console.log(kubusInhoud); // Output: 500


function berekenInhoudCilinder(diameter, hoogte) {
    const radius = diameter / 2;
    return Math.PI * Math.pow(radius, 2) * hoogte;
  }
  
  // Voorbeeld aanroep
  let cilinderInhoud = berekenInhoudCilinder(10, 20);
  console.log(cilinderInhoud); // Output: ~1570.796

 
function berekenLangeZijde(lengte, hoogte) {
    return Math.sqrt(Math.pow(lengte, 2) + Math.pow(hoogte, 2));
  }
  
  // Voorbeeld aanroep
  let langeZijde = berekenLangeZijde(3, 4);
  console.log(langeZijde); // Output: 5


function berekenGemiddelde(getal1, getal2, getal3, getal4, getal5, getal6, getal7) {
    const som = getal1 + getal2 + getal3 + getal4 + getal5 + getal6 + getal7;
    return som / 7;
  }
  
  // Voorbeeld aanroep
  let gemiddelde = berekenGemiddelde(10, 20, 30, 40, 50, 60, 70);
  console.log(gemiddelde); // Output: 40
